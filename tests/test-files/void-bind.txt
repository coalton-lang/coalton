================================================================================
2 Void function can be called for side effects
================================================================================

(package coalton-unit-tests/void-bind
  (import coalton-prelude))

(declare void-fn (Void -> Void))
(define (void-fn)
  (values))

(define (caller)
  (void-fn)
  42)

================================================================================
3 Multiple-value bind is still allowed
================================================================================

(package coalton-unit-tests/void-bind
  (import coalton-prelude))

(declare two-values (Void -> Integer * Integer))
(define (two-values)
  (values 1 2))

(define (consumer)
  (let (values a b) = (two-values))
  (+ a b))
