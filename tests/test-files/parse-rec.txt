(define x
  (rec list-length ((lst (make-list 1 2 3 4 5))
                    (len 0))

      ((Nil) len))))

3 Parse nested rec
================================================================================

(package coalton-unit-tests)

(define x
  (rec even? ((n 5))
    (if (coalton-prelude:== n 0)
        True
        (rec odd? ((m (coalton-prelude:1- n)))
          (if (coalton-prelude:== m 0)
              False
              (even? (coalton-prelude:1- m)))))))
