================================================================================
1 Parse dynamic binding
================================================================================

(package coalton-unit-tests)

(declare *x* Integer)
(define *x* 0)

(define y
  (dynamic-bind ((*x* 1))
    *x*))

================================================================================
2 Parse shorthand dynamic let in body
================================================================================

(package coalton-unit-tests)

(declare *x* Integer)
(define *x* 0)

(define y
  (let *x* = 1)
  *x*)

================================================================================
3 Parse shorthand dynamic let in do
================================================================================

(package coalton-unit-tests/prelude
  (import coalton-prelude))

(declare *x* Integer)
(define *x* 0)

(declare y (Optional Integer))
(define y
  (do
    (let *x* = 1)
    (pure *x*)))

================================================================================
100 Ordinary lexical binder cannot use earmuffs
================================================================================

(package test-package)

(define y
  (let ((*x* 1))
    *x*))

--------------------------------------------------------------------------------

error: Invalid variable
  --> test:4:9
   |
 4 |    (let ((*x* 1))
   |           ^^^ ordinary variables cannot use dynamic-variable earmuffs
